
class Animal
{
   
	void sound()
	{
		System.out.println("Animal Sound is displayed..");
	}
}
class Dog extends Animal
{
    
	void sound()
	{
		super.sound();
		System.out.println("Dogs Bark..");
	}
	void food()
	{
	    System.out.println("eats meat");
	}
}
class Horse extends Animal
{
   
	void sound()
	{
		System.out.println("Horse Neighs..");
	}
}
public class Main {

	public static void main(String[] args)
	{
		
		Animal a=new Dog();
		a.sound();
	    Dog d=new Dog();
	    d.food();
		Animal h=new Horse();
		h.sound();
		

	}

}
